class player():
    def __init__(self, name, rating):
        self.name = name
        self.score = 0
        self.games = 0
        self.rating = rating
        self.opp_ratings = []
        self.performance = 0
        
    def game(self, B, result):
        import numpy as np
        self.games += 1
        if result == self.name:
            self.score += 1
        self.opp_ratings.append(B.rating)
        
        add_wins = (self.score-(self.games/2))*2
        self.performance = np.mean(self.opp_ratings) + add_wins*15

def game(A,B):
    diff = B.rating-A.rating
    white = (1+10**(diff/400))**-1
    rng = rand()    
    if white>rng:
        return A.name
    else:
        return B.name

def swiss_tournament(players, rounds, reps):
    standings = sorted([(player.score, player.performance, player.rating, player.name) for player in players], reverse=True)
    results = [[] for i in players]
    for reps in range(reps):
        for n in range(rounds):
            for i in range(int(len(players)/2)):
                player_A = players[ord(standings[2*i][3])-65]
                player_B = players[ord(standings[2*i+1][3])-65]
                result = game(player_A, player_B)
                player_A.game(player_B, result)
                player_B.game(player_A, result)    
            standings = sorted([(player.score, player.performance, player.rating, player.name) for player in players], reverse=True)

        final_results = [d for a,b,c,d in standings]
        for n,k in enumerate(players):
            finish = final_results.index(chr(65+n))+1
            results[n].append(finish)

        for ply in players:
            ply.__init__(ply.name,ply.rating)

    bar_results = [[[],[]] for i in players]
    for i in range(len(players)):
        bar_results[i][0] = [i for i in range(1,21)]
        for j in bar_results[i][0]:
            bar_results[i][1].append(results[i].count(j))   
    return bar_results

def swiss_gambit(players,rounds,n_reps):
    sgbar_results = [[[],[]] for i in players]
    sgresults = [[] for i in players]
    standings = sorted([(player.score, player.performance, player.rating, player.name) for player in players], reverse=True)
    for en in range(len(players)):
        rem = chr(65+en)
        for reps in range(n_reps):
            for n in range(rounds):
                for i in range(int(len(players)/2)):
                    player_A = players[ord(standings[2*i][3])-65]
                    player_B = players[ord(standings[2*i+1][3])-65]
                    plys = [player_A.name,player_B.name]
                    if (n==0) & (rem in plys):
                        plys.remove(rem)
                        result = plys[0]
                    else:
                        result = game(player_A, player_B)
                    player_A.game(player_B, result)
                    player_B.game(player_A, result)    
                standings = sorted([(player.score, player.performance, player.rating, player.name) for player in players], reverse=True)
            final_results = [d for a,b,c,d in standings]
            finish = final_results.index(chr(65+en))+1
            sgresults[en].append(finish)

            for ply in players:
                ply.__init__(ply.name,ply.rating)

    for en in range(len(players)):
        sgbar_results[en][0] = [i for i in range(1,21)]
        for j in sgbar_results[en][0]:
            sgbar_results[en][1].append(sgresults[en].count(j))
    return sgbar_results
